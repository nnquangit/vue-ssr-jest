import {api} from '../helpers'
import {simpleActionGet, simpleMuta} from './helpers'
 
const actionsConst = {
  load: {
    FETCH: 'CART_LOAD_FETCH',
    FETCH_SUCCESS: 'CART_LOAD_FETCH_SUCCESS',
    FETCH_FAIL: 'CART_LOAD_FETCH_FAIL'
  },
  add: {
    FETCH: 'CART_ADD_FETCH',
    FETCH_SUCCESS: 'CART_ADD_FETCH_SUCCESS',
    FETCH_FAIL: 'CART_ADD_FETCH_FAIL'
  },
  qty: {
    FETCH: 'CART_QTY_FETCH',
    FETCH_SUCCESS: 'CART_QTY_FETCH_SUCCESS',
    FETCH_FAIL: 'CART_QTY_FETCH_FAIL'
  },
  remove: {
    FETCH: 'CART_REMOVE_FETCH',
    FETCH_SUCCESS: 'CART_REMOVE_FETCH_SUCCESS',
    FETCH_FAIL: 'CART_REMOVE_FETCH_FAIL'
  },
  /// ////////////////////
  SHIPPING: 'CART_SHIPPING',
  BILLING: 'CART_BILLING',
  PAYMENT: 'CART_PAYMENT',
  /// ////////////////////
  payments: {
    FETCH: 'CART_PAYMENTS_FETCH',
    FETCH_SUCCESS: 'CART_PAYMENTS_FETCH_SUCCESS',
    FETCH_FAIL: 'CART_PAYMENTS_FETCH_FAIL'
  }
}
 
const state = {
  fetch: false,
  data: {},
  shipping: {},
  billing: {},
  payment: {},
  payments: {
    fetch: false,
    data: []
  }
}
 
const actions = {
  cartLoad({commit, state}) {
    commit(actionsConst.load.FETCH, true)
    return api.get('/cart/load/')
      .then(res => {
        commit(actionsConst.load.FETCH_SUCCESS, res)
        return res
      })
      .catch(res => {
        commit(actionsConst.load.FETCH_FAIL, res)
        return res
      })
  },
  cartAdd({commit, state}, [id, qty = 1]) {
    commit(actionsConst.add.FETCH, true)
    return api.post('/cart/add/' + id, {qty})
      .then(res => {
        commit(actionsConst.add.FETCH_SUCCESS, res)
        return res
      })
      .catch(res => {
        commit(actionsConst.add.FETCH_FAIL, res)
        return res
      })
  },
  cartQty({commit, state}, [id, qty = 1]) {
    commit(actionsConst.qty.FETCH, true)
    return api.post('/cart/qty/' + id, {qty})
      .then(res => {
        commit(actionsConst.qty.FETCH_SUCCESS, res)
        return res
      })
      .catch(res => {
        commit(actionsConst.qty.FETCH_FAIL, res)
        return res
      })
  },
  cartRemove({commit, state}, id) {
    commit(actionsConst.remove.FETCH, true)
    return api.post('/cart/remove/' + id)
      .then(res => {
        commit(actionsConst.remove.FETCH_SUCCESS, res)
        return res
      })
      .catch(res => {
        commit(actionsConst.remove.FETCH_FAIL, res)
        return res
      })
  },
  /// ////////////////////
  cartShipping: ({commit, state}, data) => commit(actionsConst.SHIPPING, data),
  cartBilling: ({commit, state}, data) => commit(actionsConst.BILLING, data),
  cartPayment: ({commit, state}, data) => commit(actionsConst.PAYMENT, data),
  /// ////////////////////
  ...simpleActionGet('loadPayments', actionsConst.payments, '/cart/payments')
}
 
const mutations = {
  ...simpleMuta(actionsConst.load, state => state),
  ...simpleMuta(actionsConst.add, state => state),
  ...simpleMuta(actionsConst.qty, state => state),
  ...simpleMuta(actionsConst.remove, state => state),
  /// ////////////////////
  [actionsConst.SHIPPING]: (state, data) => state.shipping = data,
  [actionsConst.BILLING]: (state, data) => state.billing = data,
  [actionsConst.PAYMENT]: (state, data) => state.payment = data,
  /// ////////////////////
  ...simpleMuta(actionsConst.payments, state => state.payments)
}
 
const getters = {}
 
export default {state, actions, mutations, getters}
