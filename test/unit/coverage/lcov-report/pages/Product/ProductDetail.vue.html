    <div class="__page">
        <v-toolbar dark color="primary" tabs>
            <v-btn icon @click="$router.go(-1)">
                <i :class="icon.menu_back"></i>
            </v-btn>
            <v-spacer></v-spacer>
            <v-btn icon :loading="cart.fetch" v-if="data.id" to="/cart">
                <v-badge color="deep-orange" overlap>
                    <small slot="badge" v-if="cart.data.item_count">
                        {{cart.data.item_count}}
                    </small>
                    <v-icon :class="icon.cart"></v-icon>
                </v-badge>
            </v-btn>
            <v-btn icon :loading="fetchWishlist" v-if="data.id" @click="toogleWishlist(data.id)">
                <i :class="inWishlist(data.id)?icon.like:icon.unlike"></i>
            </v-btn>
            <v-btn icon><i :class="icon.share"></i></v-btn>
            <v-btn icon><i :class="icon.account"></i></v-btn>
            <v-tabs slot="extension" class="prod-tabs">
                <v-tabs-slider color="red"></v-tabs-slider>
                <v-tab class="text-dark" @click="onScroll('#prod-overview')">Tổng quan</v-tab>
                <v-tab class="text-dark" @click="onScroll('#prod-description')">Mô tả</v-tab>
                <v-tab class="text-dark" @click="onScroll('#prod-question')">Hỏi & đáp</v-tab>
            </v-tabs>
        </v-toolbar>
        <div class="__body" id="mainbody">
            <div id="top_page"></div>
            <div v-if="data.id">
                <div id="prod-overview" class="bg-white">
                    <ProductOverview v-if="data" v-bind="data" :title="variation.title||data.title"/>
                    <div class="__block">
                        <div class="__block__body">
                            <h2 class="__block__header">Tùy chọn sản phẩm</h2>
                            <div class="prod-vars">
                                <div class="prod-vars__inner">
                                    <div :class="'prod-vars__item '+((variation.id==e.id)?'prod-vars__item--active':'')"
                                         v-for="(e,i) in data.variations" :key="i" @click="onVariation(e)">
                                        <img :src="e.image"/>
                                    </div>
                                </div>
                            </div>
                        </div>
                    </div>
                    <ProductTags v-if="data" :tags="data.tags"/>
                </div>
                <div id="prod-description" class="__block">
                    <h2 class="__block__header">Mô tả sản phẩm</h2>
                    <div class="__block__body bg-white" v-html="data.content"></div>
                </div>
                <div id="prod-review">
                    <ProductReview v-bind="data"/>
                </div>
                <div id="prod-question">
                    <ProductFaq v-bind="data"/>
                </div>
            </div>
            <ProductRecommend/>
            <v-speed-dial v-model="fab" fixed bottom right transition="scale-transition">
                <v-btn v-model="fab" slot="activator" color="primary" dark fab hover>
                    <i :class="'icon '+icon.plus"></i>
                    <i :class="'icon '+icon.close"></i>
                </v-btn>
                <v-btn fab dark small class="mb-3" color="blue darken-2" @click="cartAdd([data.id,1])">
                    <i :class="icon.addtocart"></i>
                </v-btn>
                <v-btn fab dark small class="mb-3" color="red">
                    <i :class="icon.checkout"></i>
                </v-btn>
            </v-speed-dial>
        </div>
    </div>
</template>
<script>
  import {mapActions, mapGetters} from 'vuex'
  import {icon} from '../../constants'
 
  let asyncData = ({$store, $route, $router}) => {
    let id = parseInt($route.params.id)
    return $store.dispatch('detailProduct', id)
      .then(res => {
        if (id === 1) {
          $router.push('/404')
        }
        return res
      })
  }

  export default {
    asyncData,
    metaInfo() {
      return this.$makeMeta({
        title: this.data.title
      })
    },
    data: () => ({icon, variation: {}, fab: false}),
    mounted: function () {
      this.listReview([this.id, 1, false])
      this.listFaq([this.id, 1, false])
      this.$scrollTo('#top_page', 500, {container: '#mainbody'})
    },
    watch: {
      $route() {
        this.mounted()
      }
    },
    computed: {
      ...mapGetters(['inWishlist']),
      id() {
        return parseInt(this.$route.params.id)
      },
      fetch() {
        return this.$store.state.product.detail.fetch
      },
      fetchWishlist() {
        return this.$store.state.wishlist.toogle.fetch
      },
      data() {
        return this.$store.state.product.detail.data
      },
      cart() {
        return this.$store.state.cart
      }
    },
    methods: {
      ...mapActions(['detailProduct', 'cartAdd', 'toogleWishlist', 'listReview', 'listFaq']),
      onVariation: function (e) {
        this.variation = {...e}
      },
      onScroll: function (e) {
        this.$scrollTo(e, 500, {container: '#mainbody'})
      }
    }
  }
</script>
