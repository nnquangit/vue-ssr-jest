<script>
  import {mapActions} from 'vuex'
  import {icon} from '../../constants'
 
  export default {
    $_veeValidate: {validator: 'new'},
    data: () => ({
      icon,
      model: 'tab-info',
      /// / Thông tin cá nhân
      dd: {email: 'nnquangit@gmail.com'},
      /// / Đổi mật khẩu
      ps: {},
      sp1: true,
      sp2: true
    }),
    computed: {
      fetchProfile() {
        return this.$store.state.auth.profile.fetch
      },
      fetchPassword() {
        return this.$store.state.auth.passwd.fetch
      }
    },
    methods: {
      ...mapActions(['profile', 'passwd']),
      onProfile: function () {
        this.$validator.validateAll('dd').then(res => {
          if (res) {
            this.profile(this.dd).then(() => this.$router.push({path: '/'}))
          }
        })
      },
      onPasswd: function () {
        this.$validator.validateAll('ps').then(res => {
          if (res) {
            this.passwd(this.ps).then(() => this.$router.push({path: '/'}))
          }
        })
      }
    }
  }
</script>
