<template>
    <div class="prod__item">
        <router-link :to="link" class="prod__wrap">
            <div class="prod__thumb"><img :src="image"/></div>
            <div class="prod__body">
                <h4 class="prod__head">{{title}}</h4>
                <div class="prod__price">
                    <span>{{price|pricetext}}</span>
                    <span class="prod__origin">{{origin|pricetext}}</span>
                </div>
                <div class="prod__rate">
                    <app-rate :rate="rate"/>
                    <small class="ml-1 text-muted">({{rate_count}})</small>
                </div>
            </div>
            <div class="prod__tags"></div>
        </router-link>
        <div class="__row bg-light">
            <div class="__content px-3">Số lượng</div>
            <div class="__space">
                <v-text-field
                        color="primary" small solo flat
                        v-model="qty" @change="cartQty([id,qty])"></v-text-field>
            </div>
            <div class="__content px-2">
                <v-btn color="primary" dark icon @click="cartAdd([id,-1])">
                    <i class="mdi mdi-minus"></i>
                </v-btn>
                <v-btn color="blue" dark icon @click="cartAdd([id,1])">
                    <i class="mdi mdi-plus"></i>
                </v-btn>
                <v-btn color="red" dark icon @click="cartRemove(id)">
                    <i class="mdi mdi-delete"></i>
                </v-btn>
            </div>
        </div>
    </div>
</template>
<script>
  import {mapActions} from 'vuex'
 
  export default {
    props: ['id', 'title', 'link', 'rate', 'rate_count', 'image', 'price', 'origin', '_qty'],
    data: () => ({qty: 0}),
    mounted() {
      this.qty = this._qty
    },
    watch: {
      _qty: function (newval) {
        this.qty = newval
      }
    },
    methods: {
      ...mapActions(['cartAdd', 'cartQty', 'cartRemove'])
    }
  }
</script>
